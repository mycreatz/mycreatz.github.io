153 public static void AddTorqueFromMoment(this Ball ball, Moment moment, Vector3 normal, float touchWidth)
154 {
155 //moment: 진행 방향임
156 Vector3 expectedRotAxis = Vector3.Cross(moment.Direction, normal);
157
158 float expectedRPS = moment.Value.magnitude / moment.DeltaTime / ball.Round;
159
160 Vector3 sumRadialMoment = ball.RotationAxis * ball.RPS * (1 - ball.PresetBall.FrictionRound * BoundingTrajectoryExtension.SimulateTimeInterval)
161 + expectedRotAxis * expectedRPS;
162
163 sumRadialMoment *= (1 - ball.PresetBall.AngularDrag * touchWidth / PhysicsInfo.TouchWidthMax);
164 //Debug.Log(expectedRotAxis.ToString() + ", " + sumRadialMoment.ToString() + ", " + ball.RotationAxis.ToString());// + ", rps" + ball.RPS.ToString());
165 ball.RPS = sumRadialMoment.magnitude;
