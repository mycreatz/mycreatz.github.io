179 public static bool GetRealTimeTrajectory(this Ball targetBall, out BoundingTrajectory rst)
180 {
181 if (targetBall.BoundTimeSpent >= targetBall.TotalBoundingTime)

183 rst = targetBall.LastBoundingTrajectory;

196 BoundingTrajectory next = targetBall.BoundingTrajectories[nexIndex];

208 prev.BackSpin * (1 - interStep) + next.BackSpin * interStep, prev.SideSpin * (1 - interStep) + next.SideSpin * interStep,
